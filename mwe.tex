%-----------------------------
% (1) 定义 \datedsubsection 命令 (一个简化版)
%-----------------------------
\newcommand{\datedsubsection}[2]{
  \noindent
  {\bfseries #1} \hfill #2
  \par
}

%-----------------------------
% (2) 定义 \FancyBadge 宏
%-----------------------------
\usepackage{tikz}
\newcommand{\FancyBadge}[2]{%
  \tikz[baseline=(txt.base)]{
    \node[
      rounded corners=3pt,
      fill=#1,
      text=white,
      inner sep=2pt,
      font=\bfseries
    ] (txt) {#2};
  }%
}
